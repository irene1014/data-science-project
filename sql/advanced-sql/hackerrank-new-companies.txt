select c.company_code,c.founder,count(distinct l.lead_manager_code), count(distinct s.senior_manager_code), count(distinct m.manager_code),count(distinct e.employee_code)from company c join Lead_Manager l on  l.company_code= c.company_codejoin Senior_Manager s on  s.lead_manager_code= l.lead_manager_codejoin Manager m on  s.senior_manager_code= m.senior_manager_codejoin employee e on  m.manager_code= e.manager_codegroup by c.company_code,c.founderorder by c.company_code;
